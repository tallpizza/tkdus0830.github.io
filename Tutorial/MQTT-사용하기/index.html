    #include "Adafruit_SHT31.h"
#include <ESP8266WiFi.h>
#include <PubSubClient.h>


#define wifi_ssid "HelloWorld"
#define wifi_password "sangyeon"

#define server "192.168.10.101"
#define mqtt_user ""
#define mqtt_password ""

#define THREE_SECONDS 3000


Adafruit_SHT31 sht31 = Adafruit_SHT31();


unsigned long current_time;
unsigned long last_run = 0;
char temp[10];
char humi[10];

WiFiClient espClient;

void callback(char *topic, byte *payload, unsigned int length) {
  Serial.print("Message arrived [");
  Serial.print(topic);
  Serial.print("] ");
  for (int i = 0; i < length; i++) {
    Serial.print((char)payload[i]);
  }
  Serial.println();
}

PubSubClient client(espClient);

void reconnect() {
  // Loop until we're reconnected
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");
    // Attempt to connect
    if (client.connect("arduinoClient")) {
      Serial.println("connected");
      // Once connected, publish an announcement...
      client.publish("connected", "Welcome to Jung's Factory");
      // ... and resubscribe
      client.subscribe("connected");
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      // Wait 5 seconds before retrying
      delay(5000);
    }
  }
}

void setup() {
  Serial.begin(115200);

  client.setServer(server, 1883);
  client.setCallback(callback);
  setup_wifi();
  // Allow the hardware to sort itself out
  delay(1500);


  Serial.println("SHT31 test");
  if (!sht31.begin(0x44)) {
    Serial.println("Couldn't find SHT31");
    while (1) delay(1);
  }

}

void loop() {
  if (!client.connected()) {
    reconnect();
  }

  client.loop();
  current_time = millis();

  if (current_time - last_run >= THREE_SECONDS) {
    // save the last time you blinked the LED
    last_run = current_time;

    float t = sht31.readTemperature();
    float h = sht31.readHumidity();

    if (!isnan(t)) {  // check if 'is not a number'
      sprintf(temp, "%.2f", t);
      client.publish("mytopic/temp", temp);
      // Serial.print(" Temperature: ");
      // Serial.print(t);
      // Serial.println();
    }

    if (!isnan(h)) {  // check if 'is not a number'
      sprintf(humi, "%.2f", h);
      client.publish("mytopic/humi", humi);
      // Serial.print(" Humidity: ");
      // Serial.print(h);
      // Serial.println();
    }
  }
}


  void setup_wifi() {
    delay(10);
    // We start by connecting to a WiFi network
    Serial.println();
    Serial.print("Connecting to ");
    Serial.println(wifi_ssid);
    WiFi.begin(wifi_ssid, wifi_password);

    while (WiFi.status() != WL_CONNECTED) {
      delay(500);
      Serial.print(".");
    }

    Serial.println("");
    Serial.println("WiFi connected");
    Serial.println("IP address: ");
    Serial.println(WiFi.localIP());
  }

    import paho.mqtt.client as mqtt

# MQTT broker 정보
broker_address = "192.168.10.101"
port = 1883

# Subscriber 콜백 함수
def on_message(client, userdata, message):
    print(f"Received message: {message.topic} {str(message.payload.decode('utf-8'))}")

# MQTT client 생성
client = mqtt.Client()

# Subscriber 콜백 함수 등록
client.on_message = on_message

# MQTT broker 연결
client.connect(broker_address, port=port)

# 구독할 토픽
temp = "mytopic/temp"
humi = "mytopic/humi"

# MQTT client에 토픽 구독 요청
client.subscribe(temp)
client.subscribe(humi)

# Subscriber 유지
client.loop_forever()
